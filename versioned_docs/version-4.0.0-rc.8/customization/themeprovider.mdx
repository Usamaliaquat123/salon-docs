```tsx


const {
   loading,
    language,
    themeMode,
    accessToken,
    setReduxAccessToken,
    setReduxLanguage,
    setReduxThemeMode,
} = useHAppConfig()

console.log(loading) /** Loading state of the app **/
console.log(language) /** Language state of the app **/
console.log(themeMode) /** Current theme of the app **/
console.log(accessToken) /** User Current Token of the app **/
console.log(setReduxAccessToken) /** set the access token on the app **/
console.log(setReduxLanguage) /** set the language of the app **/
console.log(setReduxThemeMode) /** set the theme of the app **/

```

## setReduxAccessToken
Set the token of the current logged in `User`

```tsx
const { setReduxAccessToken } = useHAppConfig()

console.log(setReduxAccessToken) /** set the token of the current logged in user **/

```


## setReduxThemeMode
Set the theme of the app

```tsx
const { setReduxThemeMode } = useHAppConfig()

console.log(setReduxThemeMode) /** purple | pink **/

```
## setReduxLanguage
Set the language of the current app level language

```tsx
const { setReduxLanguage } = useHAppConfig()

console.log(setReduxLanguage) /** set the language en | ar **/

```


## accessToken
show you the current user accessToken

```tsx
const { accessToken } = useHUser()

console.log(accessToken) /** current user Access Token **/

```

## Language
language will show you the current app level language

```tsx
const { language } = useHUser()

console.log(language) /** Language state of the app **/

```

## Theme Mode
current app theme

```tsx
const { themeMode } = useHUser()

console.log(themeMode) /** Current theme of the app **/

```


## Loading
loading state of the app
```tsx
const { loading } = useHUser()

console.log(loading) /** Loading state of the app **/

```
