This component is used for user image

### Import

```tsx
import {
  Avatar as AvatarUi,
  AvatarFallback,
  AvatarImage,
} from "@/components/ui/avatar"; // shadcn ui
```

### Props

```tsx
interface Props {
  image: string;
  fallback: string;
  className?: string;
}
```

### component

```tsx
const Avatar: React.FC<Props> = ({ image, className, fallback }) => {
  return (
    <AvatarUi className={className}>
      <AvatarImage src={image}/>
      <AvatarFallback>{fallback}</AvatarFallback>
    </AvatarUi>
  );
};

export default Avatar;
```

### Usage

```tsx
<Avatar image="" fallback="" className=""/>
```
