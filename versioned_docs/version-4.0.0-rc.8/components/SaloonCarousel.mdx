### Import

```tsx
"use client";
import React, { ReactNode, type FC } from "react";
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselNext,
  CarouselPrevious,
} from "../ui/carousel";
import { Card, CardContent } from "../ui/card";

import { Saloon } from "@/components/types/salon";
import Image from "next/image";
import { TiStarFullOutline } from "react-icons/ti";
import { Badge } from "../ui/badge";
import { cn } from "@/lib/utils";
import Link from "next/link";
```

### Props

```tsx
interface SaloonCarouselProps {
  title: ReactNode;
  className?: string;
  data: Saloon[];
}
```

### component

```tsx
const SaloonCarousel: FC<SaloonCarouselProps> = ({
  title,
  className,
  data,
}) => {
  if (data.length === 0) return null;
  return (
    <div className={cn(className)}>
      <Carousel
        opts={{
          align: "start",
        }}
        className="w-[85%] md:w-full  m-auto flex flex-col gap-3"
      >
        <span className="font-bold text-2xl">{title} </span>
        <CarouselContent className="gap-3">
          {data?.map((saloon: Saloon) => (
            <Link href={`a/${saloon?.id}`}>
              <CarouselItem
                key={saloon?.id}
                className="w-[225px] h-[255px]  p-0 rounded-xl overflow-hidden border border-black/20"
              >
                <Card>
                  <CardContent className="flex flex-col bg-white w-[220px] h-[250px] justify-start  aspect-square items-start gap-3 p-0 border-0 ">
                    <Image
                      width={300}
                      height={300}
                      alt=""
                      className="aspect-square w-full h-[50%]"
                      src={saloon.picture.downloadUrl}
                    />
                    <section className="flex flex-col p-3 gap-1">
                      <span className="text-lg font-semibold">
                        {saloon?.name}
                      </span>
                      <section className="flex flex-row text-xs items-center gap-2 font-semibold">
                        5.0
                        <TiStarFullOutline />
                        <span className="font-normal ">(1,429)</span>
                      </section>
                      <span className="text-black/50 text-xs">
                        Moyfair, London
                      </span>

                      <Badge
                        variant="outline"
                        className="border-black/30 rounded text-xs"
                      >
                        Barbershop
                      </Badge>
                    </section>
                  </CardContent>
                </Card>
              </CarouselItem>
            </Link>
          ))}
        </CarouselContent>
        <CarouselPrevious />
        <CarouselNext />
      </Carousel>
    </div>
  );
};
export default SaloonCarousel;
```
