### Import

```tsx
"use client";
import React, { useState } from "react";
import CarouselBar from "../detailes/CarouselBar";
import { Swiper, SwiperSlide } from "swiper/react";
import { Navigation } from "swiper/modules";
import { popularCarousel } from "./data";
import Image from "next/image";
import { Rating, RoundedStar } from "@smastrom/react-rating";
import Link from "next/link";
```

### Props

```tsx
import { popularCarousel } from "./data";
```

### component

```tsx
export function PopularCarousel() {
    const [rating, setRating] = useState(3);

    const includedShapesStyles = [RoundedStar].map((itemShapes) => ({
        itemShapes,
        activeFillColor: "#FDD501",
        inactiveFillColor: "#F0F0F1",
    }));
    return (
        <>
            <div className="mt-10">
                <CarouselBar
                    title="Popular Shops"
                    link="#"
                    nextId="popularNextCarousel"
                    prevId="popularPrevCarousel"
                />
            </div>
            <Swiper
                modules={[Navigation]}
                spaceBetween={20}
                slidesPerView={1.5}
                breakpoints={{
                    768: {
                        slidesPerView: 2.5,
                    },
                }}
                navigation={{
                    nextEl: "#popularNextCarousel",
                    prevEl: "#popularPrevCarousel",
                }}
            >
                {popularCarousel.map((item) => (
                    <SwiperSlide key={item.name}>
                        <Link
                            href={`/a/SALOON_1709132497476`}
                            className="border border-border-primary rounded-xl p-2.5 flex gap-2.5"
                        >
                            <Image
                                src={item.image}
                                alt=""
                                width={70}
                                height={70}
                                className="rounded-xl w-auto aspect-square"
                            />
                            <div className="flex flex-col">
                                <p className="text-content-primary font-medium mb-auto">
                                    {item.name}
                                </p>
                                {includedShapesStyles.map(
                                    (itemStyles, index) => (
                                        <Rating
                                            key={`shape_${index}`}
                                            value={rating}
                                            itemStyles={itemStyles}
                                            style={{
                                                maxWidth: 100,
                                            }}
                                            readOnly
                                        />
                                    )
                                )}
                                <div className="flex items-center gap-0.5 text-brand-2 text-sm capitalize">
                                    {item.categories.join(" . ")}
                                </div>
                            </div>
                        </Link>
                    </SwiperSlide>
                ))}
            </Swiper>
        </>
    );
}

```
